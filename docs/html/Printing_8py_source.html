115 @staticmethod
116 def get_input(
117 bytes_to_be_read: int,
118 return_type: Callable = str
119 ):
120 """
121 @brief Função complexa que fará leitura de entrada do usuário
122 @details
123 Tome cuidado com a execução dessa função, pois ela é poderosa
124 @param return_type Tipo de entrada a ser retornado
125 @param bytes_to_be_read Quantidade de Bytes que serão lidos
126 @return Entrada do usuário
127 """
128
129 # Obtém o File Descriptor do stdin
130 fd = sys.stdin.fileno()
131
132 # Guarda modo original (echo, buffering, etc) para restaurar depois
133 old_settings = termios.tcgetattr(fd)
134
135 buffer = ""
136
137 try:
138 # - Desativa buffering de linha (não espera Enter)
139 # - Desativa echo (não mostra teclas na tela)
140 # - Desativa processamento de caracteres especiais (Ctrl+C, etc)
141 # - Captura teclas imediatamente
142 tty.setraw(fd)
143
144 while len(buffer) < bytes_to_be_read:
145 # Verifica se há input disponível (não-bloqueante)
146 if select([sys.stdin], [], [], 0.5)[0]:
147 # Adicionamos cada caractere
148 buffer += sys.stdin.read(1)
149 if buffer[-1] in {'\r', '\n'}:
150 break
151 finally:
152 # Restaura configurações originais do terminal
153 # Garante que o terminal volta ao normal mesmo com erros
154 termios.tcsetattr(fd, termios.TCSADRAIN, old_settings)
155
156 try:
157 return return_type(buffer)
158 except (ValueError, TypeError):
159 Printing.print_message("Erro de entrada!", "error")
160 return None
