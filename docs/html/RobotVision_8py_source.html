23 def project_point(self, triple_numbers: list[float], forward_axis="+x", flip_x=True, flip_y=False) -> list[float]:
24 ah = radians(triple_numbers[1])
25 av = radians(triple_numbers[2])
26
27 x = triple_numbers[0] * cos(av) * cos(ah)
28 y = triple_numbers[0] * cos(av) * sin(ah)
29 z = triple_numbers[0] * sin(av)
30
31 if forward_axis == '+x':
32 depth = x
33 cx = y
34 cy = z
35 elif forward_axis == '-x':
36 depth = -x
37 cx = -y
38 cy = z
39 elif forward_axis == '+z':
40 depth = z
41 cx = x
42 cy = y
43 elif forward_axis == '-z':
44 depth = -z
45 cx = x
46 cy = -y
47 else:
48 # default
49 depth = x
50 cx = y
51 cy = z
52
53 # Vamos definir x como profundidade?
54
55 fx = (self.width / 2) / tan(self.fov_h / 2)
56 fy = (self.height / 2) / tan(self.fov_v / 2)
57
58 x_ndc = (cx / depth) * (-1 if flip_x else 1) # NDC horizontal (negativo corrige espelho)
59 y_ndc = (cy / depth) * (-1 if flip_y else 1) # NDC vertical
60
61 u = fx * x_ndc + self.width / 2
62 v = - fy * y_ndc + self.height / 2
63
64 scale = 0.5 * fx / depth
65
66 return [u, v, scale]

202 # Se não houver informação visual (ex: robô cego ou mensagem de status puro)
